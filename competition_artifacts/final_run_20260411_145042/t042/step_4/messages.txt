def parse_frontmatter_simple(content):
    """Extract frontmatter fields from file content."""
    lines = content.split('\n')
    if lines[0].strip() != '---':
        return None
    fm = {}
    end_idx = None
    for i in range(1, len(lines)):
        if lines[i].strip() == '---':
            end_idx = i
            break
        if ':' in lines[i]:
            key, val = lines[i].split(':', 1)
            fm[key.strip()] = val.strip()
    if end_idx is None:
        return None
    body = '\n'.join(lines[end_idx+1:])
    return fm, body
